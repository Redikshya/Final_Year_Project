CREATE DATABASE IF NOT EXISTS lockify_db;
USE lockify_db;

CREATE TABLE IF NOT EXISTS uuser (
    id VARCHAR(20) NOT NULL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    department VARCHAR(2) NOT NULL,   
    status ENUM('registered') NOT NULL
);

CREATE TABLE IF NOT EXISTS access (
    access_id INT AUTO_INCREMENT PRIMARY KEY,
    id VARCHAR(20) NOT NULL,
    name VARCHAR(100) NOT NULL,
    access_date DATE NOT NULL,       -- store only date
    FOREIGN KEY (id) REFERENCES uuser(id) ON DELETE CASCADE
);

CREATE TABLE admin_passcode (
    passcode_id INT AUTO_INCREMENT PRIMARY KEY,
    passcode VARCHAR(255) NOT NULL,
    last_updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS files (
    file_id INT AUTO_INCREMENT PRIMARY KEY,
    file_name VARCHAR(255) NOT NULL,
    department VARCHAR(2) NOT NULL,       -- 01, 02, 03
    file_path VARCHAR(500) NOT NULL       -- Path to file on server
);
